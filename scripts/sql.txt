AND fm_products.ID_CATEGORIE = fm_products_categories.ID

     ->sont pas remplis il ne retour pas le produit
	 
SELECT fm_products.CODE, fm_products.DESCRIPTION, fm_products.SELLING_PRICE, fm_products_categories.NAME AS CATEGORIE, fm_providers.NAME AS PROVIDER, fm_products.STOCK, fm_products.STATE
FROM fm_products
LEFT OUTER JOIN fm_products_categories    
    ON fm_products.ID_CATEGORIE = fm_products_categories.ID
LEFT OUTER JOIN fm_providers
    ON fm_products.ID_PROVIDER = fm_providers.ID

select TAB_INVOICES.ID, TAB_INVOICES.IDATE, TAB_INVOICES.LIMIT_PAYMENTDATE, TAB_INVOICES.DESCRIPTION, TAB_INVOICES.CODE AS ICODE, TAB_PROPOSALS.CODE AS PCODE, TAB_INVOICES.DESCRIPTION, TAB_INVOICES.PRICE, TAB_INVOICES.STATE
